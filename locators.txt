Селекторы для поиска элементов в форме регистрации
".//label[text()='Имя']/parent::div/input"  # Поиск поля "Имя"
".//label[text()='Email']/parent::div/input"    # Поиск поля "e-mail"
".//input[@name='Пароль']"  # Поиск поля "Пароль"
".//button[text()='Зарегистрироваться']"    # Поиск кнопки "Зарегистрироваться"
".//button[text()='Войти']" # Поиск кнопки "Войти"
".//a[@href='/login']") # Поиск кнопки "Войти"
".input__error" # Поиск элемента с ошибкой
".//a[@href='/forgot-password']"    # Поиск перехода к "Восстановление пароля"
